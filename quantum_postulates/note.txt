def expectation_value(state, operator):
    return (state.conj().T @ operator @ state)[0, 0].real

def variance(state, operator):
    A2 = operator @ operator
    return expectation_value(state, A2) - expectation_value(state, operator)**2

from scipy.linalg import expm

# ====================== CONSTANTS & HELPERS ======================
hbar = 1.0
np.random.seed(42)

sigma_x = np.array([[0, 1], [1, 0]], dtype=complex)
sigma_y = np.array([[0, -1j], [1j, 0]], dtype=complex)
sigma_z = np.array([[1, 0], [0, -1]], dtype=complex)

def normalize(v):
    return v / np.linalg.norm(v)

def bloch_vector(state):
    """Return (x, y, z) Bloch coordinates from |ψ⟩ = [α, β]^T"""
    alpha, beta = state[0,0], state[1,0]
    theta = 2 * np.arccos(np.abs(alpha))
    phi = np.angle(beta) - np.angle(alpha) if abs(alpha) > 1e-12 else 0
    return np.sin(theta)*np.cos(phi), np.sin(theta)*np.sin(phi), np.cos(theta)

# ====================== POSTULATE 1: STATE ======================
def create_state(theta=np.pi/2, phi=0):
    """|ψ⟩ = cos(θ/2)|0⟩ + e^{iφ} sin(θ/2)|1⟩"""
    alpha = np.cos(theta/2)
    beta = np.exp(1j*phi) * np.sin(theta/2)
    return normalize(np.array([[alpha], [beta]], dtype=complex))

@interact(theta=FloatSlider(min=0, max=np.pi, step=0.01, value=np.pi/2),
