\section{1. Component Representation}
% ============================================================

\begin{definition}[Component representation]
Given a state $\mathbf{v}$ and a complete orthonormal basis $\{\mathbf{e}_{a,b,\dots}\}$ labeled by eigenvalues of commuting observables, the \textbf{component representation} (called the ``wave function'' in physics) is
\[
    v(a,b,\dots) \equiv \mathbf{e}_{a,b,\dots}^\dagger\,\mathbf{v}.
\]
It is the set of expansion coefficients.  It encodes the same information as $\mathbf{v}$; every observable's expectation value can be computed from it.
\end{definition}

\noindent The state can be reconstructed via
\[
    \mathbf{v} = \sum_{a,b,\dots} v(a,b,\dots)\,\mathbf{e}_{a,b,\dots}.
\]

\subsection{1.1 Examples}
\begin{itemize}[nosep]
    \item \textbf{Single two-level system}: basis $\{\mathbf{e}_0,\mathbf{e}_1\}$; component representation $v(k)$ takes two values.
    \item \textbf{Two two-level systems}: basis vectors are Kronecker products; $v(a,b)$ is a function of two discrete labels.
    \item \textbf{Particle on a line}: basis $\{\mathbf{e}_x\}$; $v(x) \in L^2(\mathbb{R})$ is a continuous complex-valued signal (see Lectures~9--10).
\end{itemize}

% ============================================================
\section{2. Expectation Values via Components}
% ============================================================

For an observable $L$ with matrix elements $L_{a'a}=\mathbf{e}_{a'}^\dagger L\mathbf{e}_a$:
\[
    \langle L \rangle = \sum_{a,a'} v^*(a')\,L_{a'a}\,v(a) = \mathbf{v}^\dagger L\mathbf{v}.
\]
This is the master formula from which all measurable predictions follow.

% ============================================================
\section{3. Composite Systems}
% ============================================================

For a composite system of subsystem $A$ (label $a$) and subsystem $B$ (label $b$), the component representation is $v(a,b)$.

A \textbf{product state} factorizes: $v(a,b) = v_A(a)\,v_B(b)$.  An \textbf{entangled state} is one where no such factorization exists.

\subsection{3.1 Subsystem observables}
Alice's observable $L$ has matrix elements $L_{a'b',\,ab} = L_{a'a}\,\delta_{b'b}$ (identity on Bob).  Bob's observable $M$: $M_{a'b',\,ab} = \delta_{a'a}\,M_{b'b}$ (identity on Alice).

\section{4. The Density Matrix (Partial Trace)}
% ============================================================

When Alice has access only to her subsystem and cannot observe Bob, she needs a reduced description.

\begin{definition}[Reduced density matrix (partial trace)]
\[
    \boxed{\rho^{(A)}_{a,a'} = \sum_b v^*(a',b)\,v(a,b).}
\]
Bob's variables have been \textbf{summed over} (``traced out'').  This is \textbf{marginalization}---summing out the degrees of freedom you cannot observe.
\end{definition}

\medskip
\noindent\textbf{Engineering analog:} The partial trace is exactly marginalization over latent variables in probability and statistics.  The correspondence is precise:

\medskip
\begin{center}
\begin{tabular}{l|l}
\textbf{Classical probability} & \textbf{Quantum} \\ \hline
Joint distribution $p(a,b)$ & Composite state $v(a,b)$ \\
Marginalization $p(a)=\sum_b p(a,b)$ & Partial trace $\rho^{(A)}_{a,a'}=\sum_b v^*(a',b)\,v(a,b)$ \\
Marginal distribution $p(a)$ & Reduced density matrix $\rho^{(A)}$ \\
Independence: $p(a,b)=p_A(a)\,p_B(b)$ & Product state: $v(a,b)=v_A(a)\,v_B(b)$ \\
Correlated joint & Entangled state
\end{tabular}
\end{center}

\subsection{4.1 Properties}
\begin{itemize}[nosep]
    \item $\rho^{(A)}$ is \textbf{self-adjoint}: $(\rho^{(A)}_{a,a'})^* = \rho^{(A)}_{a',a}$.
    \item $\operatorname{Tr}\rho^{(A)}=1$ (normalization).
    \item For any Alice observable $L$: $\langle L \rangle = \operatorname{Tr}(L\,\rho^{(A)}) = \sum_{a,a'} L_{a'a}\,\rho^{(A)}_{a,a'}$.
    \item $\rho^{(A)}$ contains \emph{all} statistical information Alice can ever extract from her subsystem alone.
\end{itemize}

\subsection{4.2 Product state}
If $v(a,b)=v_A(a)\,v_B(b)$, then
\[
    \rho^{(A)}_{a,a'} = v_A(a)\,v_A^*(a') \cdot \underbrace{\sum_b |v_B(b)|^2}_{=1} = v_A(a)\,v_A^*(a') = (\mathbf{v}_A\mathbf{v}_A^\dagger)_{a,a'}.
\]
This is a \textbf{rank-1 projector} $\rho^{(A)} = \mathbf{v}_A\mathbf{v}_A^\dagger$.  It has one eigenvalue $=1$ and all others $=0$.

\noindent\textbf{Proof:} Act with $\rho^{(A)}$ on a vector $\boldsymbol{\chi}$:
\[
    \rho^{(A)}\boldsymbol{\chi} = \mathbf{v}_A\mathbf{v}_A^\dagger\boldsymbol{\chi} = \mathbf{v}_A\cdot(\mathbf{v}_A^\dagger\boldsymbol{\chi}).
\]
If $\boldsymbol{\chi}=\mathbf{v}_A$: eigenvalue $+1$.  If $\boldsymbol{\chi}\perp\mathbf{v}_A$: eigenvalue $0$.  So $(\rho^{(A)})^2 = \rho^{(A)}$.

\medskip
\noindent\textbf{Concrete example.}  Take $\mathbf{v}_A = \mathbf{e}_0$ and $\mathbf{v}_B = \mathbf{x}_+ = \frac{1}{\sqrt{2}}(\mathbf{f}_0+\mathbf{f}_1)$.  The composite state is $\mathbf{e}_0\otimes\mathbf{x}_+$, with coefficient matrix $\Psi = \frac{1}{\sqrt{2}}\begin{pmatrix}1&1\\0&0\end{pmatrix}$.  Then:
\[
    \rho^{(A)} = \mathbf{e}_0\mathbf{e}_0^\dagger = \begin{pmatrix}1&0\\0&0\end{pmatrix}.
\]
Alice's reduced state is $\mathbf{e}_0$ regardless of which Bob state was chosen---confirming that in a product state, Alice's description does not depend on Bob.

\subsection{4.3 Entangled state}
For the singlet $\mathbf{v}_{\text{singlet}}=\frac{1}{\sqrt{2}}(\mathbf{e}_0\otimes\mathbf{f}_1-\mathbf{e}_1\otimes\mathbf{f}_0)$:
\[
    \rho^{(A)} = \frac{1}{2}\begin{pmatrix}1&0\\0&1\end{pmatrix} = \frac{1}{2}I.
\]
This is a \textbf{maximally mixed} state: both eigenvalues are $\frac{1}{2}$.

\medskip
\noindent\fbox{\parbox{\dimexpr\textwidth-2\fboxsep-2\fboxrule}{%
\textit{Alice has zero information about her subsystem alone, even though the composite state is perfectly known.  This is the defining feature of entanglement: the whole can be in a definite state while every part is maximally uncertain.}}}

\subsection{4.4 Purity and entanglement quantification}
The eigenvalues of $\rho^{(A)}$ are $\sigma_i^2$ where $\sigma_i$ are the singular values of the coefficient matrix (Lecture~6).  Since the state is normalized, $\sum_i \sigma_i^2 = 1$.  The standard scalar measure of how mixed the state is:
\[
    \boxed{\textbf{Purity:}\quad \gamma = \operatorname{Tr}(\rho^{(A)})^2 = \sum_i \sigma_i^4.}
\]
\begin{itemize}[nosep]
    \item \textbf{Product state (no entanglement):} one eigenvalue $=1$, rest $=0$.  Purity $=1$.
    \item \textbf{Maximal entanglement:} all eigenvalues equal ($=1/n$ for dimension $n$).  Purity $=1/n$.
    \item \textbf{Partial entanglement:} intermediate case.  $1/n < \gamma < 1$.
\end{itemize}

\medskip
\noindent\textbf{Von Neumann entropy.}  The standard entanglement measure for pure states of bipartite systems is the \textbf{von Neumann entropy}:
\[
    S = -\operatorname{Tr}(\rho^{(A)}\log\rho^{(A)}) = -\sum_i \sigma_i^2\log\sigma_i^2.
\]
This is exactly the Shannon entropy with the eigenvalues of $\rho^{(A)}$ playing the role of probabilities.  For the singlet: $S = -2\cdot\frac{1}{2}\log\frac{1}{2} = \log 2 = 1$ bit---the maximum for two qubits.  For a product state: $S=0$.

\subsection{4.5 General density matrices and mixed states}
So far we derived $\rho^{(A)}$ as a partial trace of a pure entangled state.  More generally, a \textbf{density matrix} is any positive semidefinite Hermitian matrix with trace~1.  It represents either:
\begin{enumerate}[nosep]
    \item[(a)] a subsystem of an entangled pure state (as above), or
    \item[(b)] a classical probabilistic mixture: $\rho = \sum_i p_i\,\mathbf{v}_i\mathbf{v}_i^\dagger$, where you don't know which pure state $\mathbf{v}_i$ the system is in.
\end{enumerate}
Remarkably, these two interpretations give the \emph{same} mathematical object.  This is the \textbf{purification theorem}: every mixed state on $A$ arises as the partial trace of some pure state on $A\otimes B$ for a suitably chosen $B$.  Classical uncertainty (don't know which pure state) and quantum entanglement (no pure state of the subsystem exists) produce \emph{identical} operational predictions for Alice's local measurements.

\section{5. Measurement as Entanglement}
% ============================================================

A measurement is the process of entangling a system with an apparatus.  Model: the apparatus has two states $\mathbf{f}_0$ (ready) and $\mathbf{f}_1$ (detected).  The measurement interaction is a \textbf{CNOT gate}---the system controls whether the apparatus flips:
\[
    U(\mathbf{e}_0\otimes\mathbf{f}_0) = \mathbf{e}_0\otimes\mathbf{f}_0, \qquad
    U(\mathbf{e}_1\otimes\mathbf{f}_0) = \mathbf{e}_1\otimes\mathbf{f}_1.
\]
The apparatus stays in $\mathbf{f}_0$ if it sees $\mathbf{e}_0$ and flips to $\mathbf{f}_1$ if it sees $\mathbf{e}_1$.  For a general state $\alpha\mathbf{e}_0+\beta\mathbf{e}_1$, by linearity:
\[
    U\bigl((\alpha\mathbf{e}_0+\beta\mathbf{e}_1)\otimes\mathbf{f}_0\bigr) = \alpha(\mathbf{e}_0\otimes\mathbf{f}_0) + \beta(\mathbf{e}_1\otimes\mathbf{f}_1).
\]
This is an \textbf{entangled state} between system and apparatus.  Including the apparatus in the quantum description, there is no ``collapse''---only the development of entanglement via unitary evolution.  The state update rule is what you see when you trace out the apparatus.

\section{6. Locality and No-Signaling}
% ============================================================

Suppose Alice and Bob share an entangled state and separate to distant locations.

\noindent\textbf{Key result:} Any operation Bob performs on his subsystem alone has \emph{no effect} on Alice's density matrix $\rho^{(A)}$.

\medskip
\noindent\textbf{Proof (unitary operations).}  Let Bob apply a unitary $U_B$, which acts on the composite system as $I\otimes U_B$.  The new composite state is $(I\otimes U_B)\mathbf{v}$.  Alice's reduced density matrix becomes:
\begin{align*}
    \operatorname{Tr}_B\bigl[(I\otimes U_B)|\mathbf{v}\rangle\langle\mathbf{v}|(I\otimes U_B^\dagger)\bigr]
    &= \operatorname{Tr}_B\bigl[|\mathbf{v}\rangle\langle\mathbf{v}|\,(I\otimes U_B^\dagger U_B)\bigr] \\
    &= \operatorname{Tr}_B\bigl[|\mathbf{v}\rangle\langle\mathbf{v}|\bigr] = \rho^{(A)},
\end{align*}
where the first step uses cyclicity of the partial trace over Bob's space (since $U_B$ acts only on Bob), and the second uses $U_B^\dagger U_B = I$.\;$\square$

\medskip
\noindent\textbf{Proof (measurements).}  If Bob measures an observable with eigenprojectors $\{P_k\}$ satisfying $\sum_k P_k = I$, the post-measurement ensemble (averaged over Bob's outcomes) gives Alice's density matrix:
\[
    \sum_k \operatorname{Tr}_B\bigl[(I\otimes P_k)|\mathbf{v}\rangle\langle\mathbf{v}|(I\otimes P_k)\bigr]
    = \operatorname{Tr}_B\bigl[|\mathbf{v}\rangle\langle\mathbf{v}|\,(I\otimes \underbrace{\textstyle\sum_k P_k}_{=I})\bigr]
    = \rho^{(A)}.
\]
So neither Bob's unitaries nor his measurements can change Alice's local statistics.  This is the \textbf{no-signaling theorem}---the fundamental reason quantum mechanics is consistent with special relativity.

\medskip
\noindent Therefore:
\begin{itemize}[nosep]
    \item Bob cannot send a message to Alice by manipulating his subsystem.
    \item All of Alice's measurement statistics remain unchanged.
\end{itemize}

\section{7. Summary}
% ============================================================

\begin{enumerate}[nosep]
    \item The component representation $v(a,b,\dots)=\mathbf{e}_{a,b,\dots}^\dagger\mathbf{v}$ is the set of expansion coefficients in a chosen basis.
    \item The \textbf{reduced density matrix} $\rho^{(A)}_{a,a'}=\sum_b v^*(a',b)\,v(a,b)$ describes a subsystem by tracing over (marginalizing out) inaccessible degrees of freedom.
    \item In a product state, $\rho^{(A)}$ is a rank-1 projector (one eigenvalue $=1$); in an entangled state, it is mixed (multiple nonzero eigenvalues).
    \item \textbf{Maximal entanglement}: all eigenvalues of $\rho^{(A)}$ are equal ($=1/n$).
    \item \textbf{Measurement is entanglement}: the apparatus becomes correlated with the system via unitary evolution.  The state update rule is the description when the apparatus is traced out.
    \item \textbf{Locality}: Bob's operations cannot change Alice's density matrix---no signaling.
\end{enumerate}
