\section{1. Combining Quantum Systems: The Kronecker Product}
% ============================================================

\begin{definition}[Kronecker product]
Given system $A$ with state space $\mathbb{C}^{d_A}$ (basis $\{\mathbf{e}_a\}$) and system $B$ with state space $\mathbb{C}^{d_B}$ (basis $\{\mathbf{f}_b\}$), the \textbf{combined} state space is
\[
    \mathbb{C}^{d_A} \otimes \mathbb{C}^{d_B} = \mathbb{C}^{d_A d_B},
\]
with dimension $d_A \times d_B$.  Its basis vectors are all Kronecker products $\mathbf{e}_a \otimes \mathbf{f}_b$.
\end{definition}

\noindent The Kronecker product $\otimes$ is the standard operation from linear algebra and DSP/controls---it constructs the composite space.

\subsection{1.1 Inner product on the composite space}
\[
    (\mathbf{e}_a \otimes \mathbf{f}_b)^\dagger\,(\mathbf{e}_{a'} \otimes \mathbf{f}_{b'}) = \delta_{aa'}\,\delta_{bb'}.
\]
Two composite basis states are orthogonal unless \emph{both} labels match.

\subsection{1.2 General state}
\[
    \mathbf{v} = \sum_{a,b} \alpha_{ab}\,(\mathbf{e}_a \otimes \mathbf{f}_b), \qquad \alpha_{ab}\in\mathbb{C}.
\]

% ============================================================
\section{2. Two Two-Level Systems}
% ============================================================

Each subsystem lives in $\mathbb{C}^2$.  The combined space is $\mathbb{C}^2 \otimes \mathbb{C}^2 = \mathbb{C}^4$ with basis:
\[
    \mathbf{e}_0\otimes\mathbf{f}_0,\quad \mathbf{e}_0\otimes\mathbf{f}_1,\quad \mathbf{e}_1\otimes\mathbf{f}_0,\quad \mathbf{e}_1\otimes\mathbf{f}_1.
\]

A general state:
\[
    \mathbf{v} = \alpha_{00}\,(\mathbf{e}_0\otimes\mathbf{f}_0)
    + \alpha_{01}\,(\mathbf{e}_0\otimes\mathbf{f}_1)
    + \alpha_{10}\,(\mathbf{e}_1\otimes\mathbf{f}_0)
    + \alpha_{11}\,(\mathbf{e}_1\otimes\mathbf{f}_1).
\]

% ============================================================
\section{3. Entanglement as a Rank Condition}
% ============================================================

Write the four coefficients as a \textbf{coefficient matrix}:
\[
    \Psi = \begin{pmatrix}\alpha_{00} & \alpha_{01} \\ \alpha_{10} & \alpha_{11}\end{pmatrix}.
\]

\begin{definition}[Product state]
A state is a \textbf{product state} (unentangled) if $\operatorname{rank}(\Psi) = 1$, i.e., $\Psi = \mathbf{a}\mathbf{b}^\top$ for some $\mathbf{a}, \mathbf{b} \in \mathbb{C}^2$.  Equivalently, the coefficients factorize: $\alpha_{ab} = a_a\,b_b$.
\end{definition}

\begin{definition}[Entangled state]
A state is \textbf{entangled} if $\operatorname{rank}(\Psi) \geq 2$.  It cannot be written as a Kronecker product.
\end{definition}

\subsection{3.1 Parameter counting}
A general two-system state has 4 complex coefficients $= 8$ real parameters.  Subtract 1 for normalization and 1 for overall phase: \textbf{6 real parameters}.

A product state is specified by one single-system state (2 real params) for $A$ and one (2 real params) for $B$: \textbf{4 real parameters}.

Since $6>4$, the product states form a \emph{lower-dimensional} subset: \textbf{most states are entangled}.

\subsection{3.2 Schmidt decomposition (SVD of the coefficient matrix)}
The \textbf{Schmidt decomposition} is the SVD of $\Psi$:
\[
    \Psi = U\,\Sigma\,V^\dagger, \qquad \Sigma = \operatorname{diag}(\sigma_1, \sigma_2).
\]
The singular values $\sigma_i$ quantify the degree of entanglement:
\begin{itemize}[nosep]
    \item \textbf{Product state}: one nonzero singular value ($\operatorname{rank} = 1$).
    \item \textbf{Maximally entangled}: equal singular values ($\sigma_1 = \sigma_2 = 1/\sqrt{2}$).
    \item \textbf{Partially entangled}: unequal nonzero singular values.
\end{itemize}

\medskip
\noindent\textbf{Engineering analog:} The Schmidt decomposition \emph{is} the SVD of the coefficient matrix---the same tool used throughout signal processing and data analysis.

\subsection{3.3 Example: product state}
$\mathbf{e}_0 \otimes \mathbf{x}_+ = \frac{1}{\sqrt{2}}\bigl(\mathbf{e}_0\otimes\mathbf{f}_0+\mathbf{e}_0\otimes\mathbf{f}_1\bigr)$.  Coefficient matrix:
$\Psi = \begin{pmatrix}1/\sqrt{2} & 1/\sqrt{2} \\ 0 & 0\end{pmatrix}$, which has rank 1.

\subsection{3.4 Example: maximally entangled state (singlet)}
\[
    \mathbf{v}_{\text{singlet}} = \frac{1}{\sqrt{2}}\bigl(\mathbf{e}_0\otimes\mathbf{f}_1 - \mathbf{e}_1\otimes\mathbf{f}_0\bigr).
\]
Coefficient matrix:
\[
    \Psi = \frac{1}{\sqrt{2}}\begin{pmatrix}0 & 1 \\ -1 & 0\end{pmatrix}.
\]
\textbf{Rank check via determinant:}
\[
    \det(\Psi) = \frac{1}{2}\bigl(0\cdot 0 - 1\cdot(-1)\bigr) = \frac{1}{2} \neq 0
    \quad\Longrightarrow\quad \operatorname{rank}(\Psi)=2.
\]
This cannot be written as a Kronecker product---the state is entangled.

\textbf{SVD check:}  The singular values of $\Psi$ are $\sigma_1 = \sigma_2 = \frac{1}{\sqrt{2}}$.  Equal singular values means \emph{maximally entangled} by the criterion in \S3.2.  (Verify: $\Psi^\dagger\Psi = \frac{1}{2}I$, so both singular values are $1/\sqrt{2}$.)

% ============================================================
\section{4. Observables on Composite Systems}
% ============================================================

\subsection{4.1 Subsystem observables}
An observable $L$ belonging to subsystem $A$ acts as $L \otimes I$ on the composite space (identity on $B$).  Similarly, Bob's observable $M$ acts as $I \otimes M$.

\subsection{4.2 Expectation values in a product state}
In a product state $\mathbf{v} = \mathbf{a} \otimes \mathbf{b}$:
\[
    \langle L \otimes I \rangle = \mathbf{a}^\dagger L\mathbf{a}, \qquad
    \langle I \otimes M \rangle = \mathbf{b}^\dagger M\mathbf{b}.
\]
Alice's expectation values are the same as if Bob did not exist, and vice versa.

\subsection{4.3 Uncorrelated observables in product states}
For a product state:
\[
    \langle L \otimes M \rangle = (\mathbf{a}^\dagger L\mathbf{a})(\mathbf{b}^\dagger M\mathbf{b}) = \langle L \rangle\,\langle M \rangle.
\]
The variables are \textbf{uncorrelated}.

% ============================================================
\section{5. Correlations and Entanglement}
% ============================================================

\begin{definition}[Correlation]
The \textbf{correlation} between Alice's observable $L$ and Bob's observable $M$ is
\[
    C(L,M) = \langle L \otimes M \rangle - \langle L \rangle\langle M \rangle.
\]
\end{definition}

\begin{itemize}[nosep]
    \item For product states: $C(L,M)=0$ for all $L,M$.
    \item For entangled states: there exists at least one pair $(L,M)$ with $C(L,M)\neq 0$.
\end{itemize}

\subsection{5.1 Singlet state correlations}
In the singlet $\mathbf{v}_{\text{singlet}}$, all individual expectation values vanish: $\langle Z \otimes I \rangle = 0$, $\langle X \otimes I \rangle = 0$, $\langle Y \otimes I \rangle = 0$.

But for joint measurements of matching observables:
\[
    \langle Z \otimes Z \rangle = -1, \qquad \langle X \otimes X \rangle = -1, \qquad \langle Y \otimes Y \rangle = -1.
\]
\textbf{Calculation of $\langle Z \otimes Z \rangle$}: $Z$ acts on subsystem $A$ and $Z$ acts on subsystem $B$.  On $\mathbf{e}_0\otimes\mathbf{f}_1$: eigenvalues $(+1)(-1)=-1$.  On $\mathbf{e}_1\otimes\mathbf{f}_0$: eigenvalues $(-1)(+1)=-1$.  Both terms give $-1$, so $\langle Z \otimes Z \rangle = -1$.

\textbf{Calculation of $\langle X \otimes X \rangle$}: Apply $X\otimes X$ directly to the singlet.  Since $X\mathbf{e}_0 = \mathbf{e}_1$ and $X\mathbf{e}_1 = \mathbf{e}_0$:
\begin{align*}
    (X\otimes X)\mathbf{v}_{\text{singlet}}
    &= \tfrac{1}{\sqrt{2}}\bigl(X\mathbf{e}_0\otimes X\mathbf{f}_1 - X\mathbf{e}_1\otimes X\mathbf{f}_0\bigr) \\
    &= \tfrac{1}{\sqrt{2}}\bigl(\mathbf{e}_1\otimes\mathbf{f}_0 - \mathbf{e}_0\otimes\mathbf{f}_1\bigr)
    = -\mathbf{v}_{\text{singlet}}.
\end{align*}
So $\mathbf{v}_{\text{singlet}}$ is an eigenvector of $X\otimes X$ with eigenvalue $-1$, hence $\langle X\otimes X\rangle = -1$.

\textbf{Calculation of $\langle Y \otimes Y \rangle$}: Similarly, $Y\mathbf{e}_0 = i\mathbf{e}_1$ and $Y\mathbf{e}_1 = -i\mathbf{e}_0$:
\begin{align*}
    (Y\otimes Y)\mathbf{v}_{\text{singlet}}
    &= \tfrac{1}{\sqrt{2}}\bigl(i\mathbf{e}_1\otimes(-i)\mathbf{f}_0 - (-i)\mathbf{e}_0\otimes i\mathbf{f}_1\bigr) \\
    &= \tfrac{1}{\sqrt{2}}\bigl(\mathbf{e}_1\otimes\mathbf{f}_0 - \mathbf{e}_0\otimes\mathbf{f}_1\bigr)
    = -\mathbf{v}_{\text{singlet}}.
\end{align*}
So $\langle Y\otimes Y\rangle = -1$ as well.

\medskip
\noindent\textbf{Rotational invariance of the singlet.}  The singlet is a \emph{simultaneous eigenvector} of $X\otimes X$, $Y\otimes Y$, and $Z\otimes Z$, all with eigenvalue $-1$.  This is the algebraic expression of its rotational invariance: the singlet is the unique (up to phase) state in $\mathbb{C}^2\otimes\mathbb{C}^2$ that is invariant under simultaneous rotations of both subsystems---which is why it is called a ``singlet'' (spin~0, scalar under rotations).

\medskip
Correlation: $C(Z,Z) = -1 - (0)(0) = -1 \neq 0$.  Cross-products like $\langle X \otimes Z \rangle = 0$.

\medskip
\noindent\textbf{Physical meaning:} If you measure the \emph{same} observable on both subsystems, the results are always opposite.  If you measure \emph{different} observables, there is no correlation.

\subsection{5.2 Forward reference: the CHSH inequality and Bell's theorem}
The singlet correlations computed above are \emph{all the ingredients} for the most important experimental result in the foundations of quantum mechanics: the violation of Bell's inequality.  In a future lecture, we will choose four observables ($Z$, $X$ for Alice, and two rotated observables for Bob), compute the CHSH quantity $\langle A_0 B_0\rangle + \langle A_0 B_1\rangle + \langle A_1 B_0\rangle - \langle A_1 B_1\rangle$ in the singlet, and obtain $2\sqrt{2}$---the Tsirelson bound.  Any local hidden variable theory bounds this quantity by~$2$.  The singlet violates it.  That single calculation, using only the machinery of these six lectures and a $4\times 4$ matrix, proves that the universe is not classical.

\section{6. The State Update Rule Revisited}
% ============================================================

Between measurements, the state evolves unitarily via the Schr\"odinger ODE.  At measurement of observable $L$:

\begin{enumerate}[nosep]
    \item The system is in state $\mathbf{v}=\sum_i \alpha_i\mathbf{e}_i$ (expanded in eigenvectors of $L$).
    \item Measurement yields eigenvalue $\lambda_k$ with probability $|\alpha_k|^2$.
    \item Immediately after, the state resets to $\mathbf{e}_k$ (Bayesian conditioning).
\end{enumerate}

If we treat the measuring apparatus as a quantum system, then the combined system evolves unitarily---the state update emerges naturally from entanglement between the system and apparatus.

\subsection{6.1 The von Neumann measurement model}
Take a system in state $\alpha\mathbf{e}_0+\beta\mathbf{e}_1$ and an apparatus initially in state $\mathbf{f}_0$ (the ``ready'' state).  Postulate a unitary interaction $U$ that copies the system's eigenstate into the apparatus pointer:
\[
    U(\mathbf{e}_0\otimes\mathbf{f}_0) = \mathbf{e}_0\otimes\mathbf{f}_0, \qquad
    U(\mathbf{e}_1\otimes\mathbf{f}_0) = \mathbf{e}_1\otimes\mathbf{f}_1.
\]
By linearity:
\[
    U\bigl((\alpha\mathbf{e}_0+\beta\mathbf{e}_1)\otimes\mathbf{f}_0\bigr)
    = \alpha(\mathbf{e}_0\otimes\mathbf{f}_0) + \beta(\mathbf{e}_1\otimes\mathbf{f}_1).
\]
This is an entangled state (rank 2 whenever both $\alpha,\beta\neq 0$).  The system and apparatus are now correlated: if you ``look at'' the apparatus and find it in state $\mathbf{f}_0$, the system has ``collapsed'' to $\mathbf{e}_0$; if you find $\mathbf{f}_1$, the system is in $\mathbf{e}_1$.  The ``collapse'' is just the conditional state given an observation of the entangled partner---Bayesian conditioning on a correlated variable.

\medskip
\noindent A real apparatus has $\sim 10^{23}$ degrees of freedom.  Those degrees of freedom decohere the superposition on timescales far shorter than anything we can observe, which is the modern \textbf{decoherence} picture of measurement.  This motivates the density-matrix formalism (Lecture~7).

\section{7. Summary}
% ============================================================

\begin{enumerate}[nosep]
    \item Composite systems are described by the \textbf{Kronecker product} of individual state spaces ($\dim = d_A \times d_B$).
    \item The coefficient matrix $\Psi$ encodes the composite state.  $\operatorname{rank}(\Psi) = 1$ $\Leftrightarrow$ product state; $\operatorname{rank}(\Psi) \geq 2$ $\Leftrightarrow$ entangled.
    \item The \textbf{Schmidt decomposition} is the SVD of $\Psi$; the singular values quantify entanglement.
    \item In a product state, subsystem observables are uncorrelated: $\langle L \otimes M \rangle = \langle L \rangle\langle M \rangle$.
    \item Nonzero correlations $C(L,M)\neq 0$ signal entanglement.
    \item The singlet is maximally entangled: all individual expectations vanish, but $\langle M \otimes M \rangle = -1$ for $M \in \{X,Y,Z\}$.
    \item The state update rule emerges from entanglement between system and apparatus---it is not a separate law.
\end{enumerate}
