\section{1. Motivation}
% ============================================================

These notes develop quantum mechanics from the ground up using three mathematical prerequisites: complex-valued linear algebra (vectors, matrices, eigendecomposition), first-order ODEs, and basic probability.  Analogies from control theory (LTI systems, state-space models) and signal processing (Fourier transforms, bandwidth-duration product) are used freely.  No physics background is required.

The central insight is that the state of a physical system is described by a unit-norm column vector in a complex vector space, and measurable quantities correspond to self-adjoint matrices acting on that space.  Everything else---probability rules, dynamics, uncertainty relations, entanglement---follows from linear algebra.

% ============================================================
\section{2. Axioms of the Complex State Space}
% ============================================================

The entire framework rests on four axioms.  These are the \emph{complete} rule set---everything else follows from linear algebra.

\begin{axiom}[State]
The state of a system is a normalized column vector $\mathbf{v}$ in a complex vector space ($\mathbf{v}^\dagger\mathbf{v} = 1$).  For a two-level system, $\mathbf{v} \in \mathbb{C}^2$.
\end{axiom}

\begin{axiom}[Observable]
A measurable quantity is represented by a self-adjoint matrix $M = M^\dagger$.  Its eigenvalues $\lambda_i \in \mathbb{R}$ are the only values a measurement can return.
\end{axiom}

\begin{axiom}[Measurement --- Born Rule]
Measuring $M$ on a system in state $\mathbf{v}$ returns eigenvalue $\lambda_i$ with probability $P(\lambda_i) = |\mathbf{e}_i^\dagger\mathbf{v}|^2$, where $\mathbf{e}_i$ is the normalized eigenvector corresponding to $\lambda_i$.
\end{axiom}

\begin{axiom}[Time Evolution]
An isolated system evolves according to the linear ODE
\[
    i\hbar\,\dot{\mathbf{v}}(t) = H\,\mathbf{v}(t),
\]
where $H = H^\dagger$ is the system matrix (called the Hamiltonian in physics).  Solution: $\mathbf{v}(t) = e^{-iHt/\hbar}\,\mathbf{v}(0)$.
\end{axiom}

\medskip
\noindent\textbf{Derivation note (norm preservation $\Rightarrow$ unitarity $\Rightarrow$ Hermitian generator).}  Hermiticity of $H$ is not a separate assumption---it follows from Axiom~4 plus norm preservation.  If $\mathbf{v}^\dagger\mathbf{v} = 1$ must hold at all times, then $U(t) = e^{-iHt/\hbar}$ must be unitary ($U^\dagger U = I$), which requires $H = H^\dagger$.  Present in this order: norm preservation $\Rightarrow$ unitarity $\Rightarrow$ Hermitian generator.

\section{3. Measurement: State Update and Non-Commutativity}
% ============================================================

\subsection{3.1 State update rule}
Upon measuring $M$ and obtaining eigenvalue $\lambda_i$, the state resets:
\[
    \mathbf{v} \leftarrow \mathbf{e}_i.
\]
This is Bayesian conditioning: you observed an outcome, so you update to the state consistent with that outcome.  The update is instantaneous and deterministic given the result.

\medskip
\noindent\fbox{\parbox{\dimexpr\linewidth-2\fboxsep-2\fboxrule}{%
\textbf{Experiment says:} When a two-level system is prepared and immediately re-measured with the same observable, the result is always the same.  \textit{(Motivates: state update rule---measuring $M$ and obtaining $\lambda_i$ resets the state to $\mathbf{e}_i$, so the next measurement of $M$ is deterministic.)}}}

\subsection{3.2 Non-commutativity}
Sequential measurements of \emph{different} observables are order-dependent.  This has no classical probability analog.

\medskip
\noindent\textbf{Example.}  Let $Z = \begin{pmatrix}1&0\\0&-1\end{pmatrix}$ and $X = \begin{pmatrix}0&1\\1&0\end{pmatrix}$.  Prepare $\mathbf{v} = \mathbf{e}_0 = \begin{pmatrix}1\\0\end{pmatrix}$ (eigenvector of $Z$, eigenvalue $+1$).

\medskip
\noindent\textit{Path A:} Measure $Z$ $\to$ get $+1$ with certainty $\to$ state stays $\mathbf{e}_0$ $\to$ measure $Z$ again $\to$ get $+1$ with certainty.

\medskip
\noindent\textit{Path B:} Measure $X$ $\to$ get $\pm 1$ each with probability $\tfrac{1}{2}$ (since $|\mathbf{e}_{\pm}^\dagger \mathbf{e}_0|^2 = \tfrac{1}{2}$) $\to$ state resets to eigenvector of $X$ $\to$ measure $Z$ $\to$ result is now random.

\medskip
The act of measuring $X$ destroyed the definite value of $Z$.  This happens because $[Z, X] = ZX - XZ \neq 0$, so they share no common eigenbasis.  Any state that is an eigenvector of $Z$ is a superposition in the eigenbasis of $X$, and vice versa.

\medskip
\noindent\fbox{\parbox{\dimexpr\linewidth-2\fboxsep-2\fboxrule}{%
\textbf{Experiment says:} Measuring one observable of a two-level system randomizes the other.  A system prepared with definite $Z$-value gives $\pm 1$ with equal probability when $X$ is measured.  \textit{(Motivates: $[Z, X] \neq 0$---no shared eigenbasis, so a $Z$-eigenstate is an equal superposition in the $X$-eigenbasis.)}}}

\medskip
\noindent\textbf{The general rule:} Two observables $M$ and $N$ can be simultaneously measured (have simultaneous definite values) if and only if $[M, N] = 0$.  This is a pure linear algebra statement: commutativity $\Leftrightarrow$ shared eigenbasis.

\section{4. The Two-Level System}
% ============================================================

\begin{definition}[Two-level system]
A \textbf{two-level system} is a quantum system whose state space is $\mathbb{C}^2$.  Whenever any observable is measured, exactly one of two eigenvalues is returned.  (In physics this is called a ``qubit'' or ``spin-$\tfrac{1}{2}$.'')
\end{definition}

\noindent The standard basis vectors are
\[
    \mathbf{e}_0 = \begin{pmatrix}1\\0\end{pmatrix}, \qquad
    \mathbf{e}_1 = \begin{pmatrix}0\\1\end{pmatrix}.
\]
Any state can be written as $\mathbf{v} = \alpha_0\,\mathbf{e}_0 + \alpha_1\,\mathbf{e}_1$ with $|\alpha_0|^2 + |\alpha_1|^2 = 1$.

\section{5. Complex Vector Spaces}
% ============================================================

\subsection{5.1 Definition}
A \textbf{complex vector space} is a collection of objects (vectors) satisfying:
\begin{enumerate}[nosep]
    \item \textbf{Addition}: For any two vectors $\mathbf{u}$, $\mathbf{v}$, there is a sum $\mathbf{u}+\mathbf{v}=\mathbf{w}$ which is also in the space.
    \item \textbf{Commutativity}: $\mathbf{u}+\mathbf{v}=\mathbf{v}+\mathbf{u}$.
    \item \textbf{Scalar multiplication}: For any complex number $z\in\mathbb{C}$ and any vector $\mathbf{v}$, the product $z\mathbf{v}$ is also in the space.
    \item \textbf{Zero vector}: There exists a unique $\mathbf{0}$ such that $\mathbf{v}+\mathbf{0}=\mathbf{v}$ for all $\mathbf{v}$.
\end{enumerate}
Any collection of objects satisfying these rules is a vector space.

\subsection{5.2 Examples of vector spaces}
\begin{enumerate}[nosep]
    \item \textbf{Complex numbers} $\mathbb{C}$: the simplest complex vector space (one-dimensional).  Addition and scalar multiplication are just ordinary complex arithmetic.
    \item \textbf{Column vectors} $\mathbb{C}^N$: a column of $N$ complex entries.  Addition and scalar multiplication act entry-by-entry.
    \item \textbf{Complex-valued functions} $f(x) \in L^2(\mathbb{R})$: you can add two functions, and you can multiply a function by a complex number.  This is an \emph{infinite}-dimensional vector space.
\end{enumerate}

\subsection{5.3 Concrete representation: columns}
For a two-dimensional complex vector space:
\[
    \mathbf{v} = \begin{pmatrix} \alpha_1 \\ \alpha_2 \end{pmatrix}, \qquad
    \alpha_1, \alpha_2 \in \mathbb{C}.
\]
Addition and scalar multiplication:
\[
    \begin{pmatrix}\alpha_1\\\alpha_2\end{pmatrix} + \begin{pmatrix}\beta_1\\\beta_2\end{pmatrix}
    = \begin{pmatrix}\alpha_1+\beta_1\\\alpha_2+\beta_2\end{pmatrix}, \qquad
    z\begin{pmatrix}\alpha_1\\\alpha_2\end{pmatrix}
    = \begin{pmatrix}z\alpha_1\\z\alpha_2\end{pmatrix}.
\]
Columns of length 2 and columns of length 3 belong to \emph{different} vector spaces; they cannot be added.

\subsection{5.4 Conjugate transpose}
For every column vector $\mathbf{v}$ there is a corresponding row vector $\mathbf{v}^\dagger$ obtained by transposing and complex conjugating:
\[
    \mathbf{v}^\dagger = \begin{pmatrix} \alpha_1^* & \alpha_2^* \end{pmatrix}.
\]
The rules:
\begin{align}
    (\mathbf{u}+\mathbf{v})^\dagger &= \mathbf{u}^\dagger+\mathbf{v}^\dagger, \\
    (z\mathbf{v})^\dagger &= z^*\mathbf{v}^\dagger.
\end{align}
Note the complex conjugation of $z$ in the second rule---this is crucial.

\subsection{5.5 Inner product}
The standard Hermitian inner product pairs a row vector with a column vector to produce a complex number:
\[
    \mathbf{u}^\dagger\mathbf{v} = \beta_1^*\alpha_1 + \beta_2^*\alpha_2.
\]

\noindent\textbf{Key properties:}
\begin{enumerate}[nosep]
    \item \textbf{Conjugate symmetry}: $\mathbf{u}^\dagger\mathbf{v} = (\mathbf{v}^\dagger\mathbf{u})^*$.  Swapping the two vectors conjugates the result.
    \item \textbf{Squared length}: $\mathbf{v}^\dagger\mathbf{v} = |\alpha_1|^2+|\alpha_2|^2 \geq 0$, always real and non-negative.  It defines the \textbf{norm}:
    \[
        \|\mathbf{v}\| = \sqrt{\mathbf{v}^\dagger\mathbf{v}}.
    \]
    \item \textbf{Positivity}: $\mathbf{v}^\dagger\mathbf{v}=0$ if and only if $\mathbf{v}=\mathbf{0}$.
\end{enumerate}

\subsection{5.6 Orthogonality and dimension}
Two vectors are \textbf{orthogonal} if $\mathbf{u}^\dagger\mathbf{v}=0$.  Since $\mathbf{u}^\dagger\mathbf{v}=(\mathbf{v}^\dagger\mathbf{u})^*$, orthogonality is symmetric.

\medskip
\noindent\textbf{Example:}
\[
    \mathbf{e}_0=\begin{pmatrix}1\\0\end{pmatrix}, \qquad
    \mathbf{e}_1=\begin{pmatrix}0\\1\end{pmatrix}.
\]
Then $\mathbf{e}_0^\dagger\mathbf{e}_1=1^*\cdot 0 + 0^*\cdot 1 = 0$, so $\mathbf{e}_0$ and $\mathbf{e}_1$ are orthogonal.

\medskip
The \textbf{dimension} of a vector space is the maximum number of mutually orthogonal \emph{nonzero} vectors.  In a two-dimensional space, once you have found two orthogonal vectors, no third nonzero vector can be orthogonal to both.

\medskip
\noindent\textit{For a two-level system, the state space is a \textbf{two-dimensional} complex vector space $\mathbb{C}^2$.  Next time we will determine the eigenvectors of the three standard observables and their explicit column-vector forms.}
