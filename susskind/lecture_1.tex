\section{1. Motivation}
% ============================================================

These notes develop quantum mechanics from the ground up using three mathematical prerequisites: complex-valued linear algebra (vectors, matrices, eigendecomposition), first-order ODEs, and basic probability.  Analogies from control theory (LTI systems, state-space models) and signal processing (Fourier transforms, bandwidth-duration product) are used freely.  No physics background is required.

The central insight is that the state of a physical system is described by a unit-norm column vector in a complex vector space, and measurable quantities correspond to self-adjoint matrices acting on that space.  Everything else---probability rules, dynamics, uncertainty relations, entanglement---follows from linear algebra.

% ============================================================
\section{2. Axioms of the Complex State Space}
% ============================================================

The entire framework rests on four axioms.  These are the \emph{complete} rule set---everything else follows from linear algebra.

\begin{axiom}[State]
The state of a system is a normalized column vector $\mathbf{v}$ in a complex vector space ($\mathbf{v}^\dagger\mathbf{v} = 1$).  For a two-level system, $\mathbf{v} \in \mathbb{C}^2$.
\end{axiom}

\medskip
\noindent\textbf{Global phase invariance.}  The states $\mathbf{v}$ and $e^{i\varphi}\mathbf{v}$ (for any real $\varphi$) represent the \emph{same} physical state: every Born-rule probability $|\mathbf{e}_i^\dagger\mathbf{v}|^2$ and every expectation value $\mathbf{v}^\dagger M\mathbf{v}$ is unchanged by the phase.  This removes one real degree of freedom from the state description.

\begin{axiom}[Observable]
A measurable quantity is represented by a self-adjoint matrix $M = M^\dagger$.  Its eigenvalues $\lambda_i \in \mathbb{R}$ are the only values a measurement can return.
\end{axiom}

\begin{axiom}[Measurement --- Born Rule]
Measuring $M$ on a system in state $\mathbf{v}$ returns eigenvalue $\lambda_i$ with probability $P(\lambda_i) = |\mathbf{e}_i^\dagger\mathbf{v}|^2$, where $\mathbf{e}_i$ is the normalized eigenvector corresponding to $\lambda_i$.
\end{axiom}

\begin{axiom}[Time Evolution]
An isolated system evolves according to the linear ODE
\[
    i\,\dot{\mathbf{v}}(t) = H\,\mathbf{v}(t),
\]
where $H = H^\dagger$ is the system matrix (called the Hamiltonian in physics).  Solution: $\mathbf{v}(t) = e^{-iHt}\,\mathbf{v}(0)$.

\medskip
\noindent\textit{Units:} We set $\hbar = 1$ throughout.  This is a choice of units (energy measured in radians per second) that physicists make routinely; $\hbar$ is restored only when numerical SI values are needed.

\medskip
\noindent\textit{LTI analogy:} This is a linear time-invariant system with system matrix $A = -iH$.  The state-transition matrix is $\Phi(t) = e^{At} = e^{-iHt}$---the initial-condition response of a first-order vector ODE $\dot{\mathbf{v}} = A\mathbf{v}$.  In Python: \texttt{scipy.linalg.expm(-1j * H * t)}.
\end{axiom}

\subsection{2.1 Why $H$ must be Hermitian}

Hermiticity of $H$ is \emph{not} a separate assumption---it follows from Axiom~1 (normalization) plus Axiom~4 (linear evolution):
\begin{enumerate}[nosep]
    \item \textbf{Norm preservation.}  Axiom~1 requires $\mathbf{v}^\dagger\mathbf{v} = 1$ at all times.
    \item \textbf{Unitarity.}  The evolution operator $U(t) = e^{-iHt}$ must satisfy $U^\dagger(t)\,U(t) = I$.
    \item \textbf{Hermitian generator.}  Differentiating $U^\dagger U = I$ at $t = 0$ gives $H^\dagger = H$.
\end{enumerate}
This is the same argument that forces the system matrix of any norm-preserving linear flow to be skew-Hermitian: $\dot{\mathbf{v}} = A\mathbf{v}$ preserves $\|\mathbf{v}\|$ iff $A + A^\dagger = 0$, and $A = -iH$ gives $H = H^\dagger$.

\section{3. Measurement: State Update and Non-Commutativity}
% ============================================================

\subsection{3.1 State update rule}
Upon measuring $M$ and obtaining eigenvalue $\lambda_i$, the state resets:
\[
    \mathbf{v} \leftarrow \mathbf{e}_i.
\]
This is Bayesian conditioning: you observed an outcome, so you update to the state consistent with that outcome.  The update is instantaneous and deterministic given the result.

\medskip
\noindent\fbox{\parbox{\dimexpr\linewidth-2\fboxsep-2\fboxrule}{%
\textbf{Code says:} \texttt{v = e0; prob = abs(e0.conj().T @ v)**2} $\to$ \texttt{1.0}.  After collapsing to $\mathbf{e}_0$, the probability of re-obtaining the same eigenvalue is exactly~1.  \textit{(Verify: set \texttt{v = e0}, recompute all Born-rule probabilities.)}}}

\subsection{3.2 Non-commutativity}
Sequential measurements of \emph{different} observables are order-dependent.  This has no classical probability analog.

\medskip
\noindent\textbf{Example.}  Let $Z = \begin{pmatrix}1&0\\0&-1\end{pmatrix}$ and $X = \begin{pmatrix}0&1\\1&0\end{pmatrix}$.  Prepare $\mathbf{v} = \mathbf{e}_0 = \begin{pmatrix}1\\0\end{pmatrix}$ (eigenvector of $Z$, eigenvalue $+1$).

\medskip
\noindent\textit{Path A:} Measure $Z$ $\to$ get $+1$ with certainty $\to$ state stays $\mathbf{e}_0$ $\to$ measure $Z$ again $\to$ get $+1$ with certainty.

\medskip
\noindent\textit{Path B:} The eigenvectors of $X$ are $\mathbf{e}_+ = \tfrac{1}{\sqrt{2}}\begin{pmatrix}1\\1\end{pmatrix}$ and $\mathbf{e}_- = \tfrac{1}{\sqrt{2}}\begin{pmatrix}1\\-1\end{pmatrix}$ (eigenvalues $+1$ and $-1$).  Computing the Born-rule probabilities:
\[
    |\mathbf{e}_+^\dagger\mathbf{e}_0|^2 = \left|\frac{1}{\sqrt{2}}\begin{pmatrix}1&1\end{pmatrix}\begin{pmatrix}1\\0\end{pmatrix}\right|^2 = \frac{1}{2}, \qquad
    |\mathbf{e}_-^\dagger\mathbf{e}_0|^2 = \frac{1}{2}.
\]
Measure $X$ $\to$ get $\pm 1$ each with probability $\tfrac{1}{2}$ $\to$ state resets to $\mathbf{e}_+$ or $\mathbf{e}_-$ $\to$ measure $Z$ $\to$ result is now random.

\medskip
The act of measuring $X$ destroyed the definite value of $Z$.  This happens because $[Z, X] = ZX - XZ \neq 0$, so they share no common eigenbasis.  Any state that is an eigenvector of $Z$ is a superposition in the eigenbasis of $X$, and vice versa.

\medskip
\noindent\fbox{\parbox{\dimexpr\linewidth-2\fboxsep-2\fboxrule}{%
\textbf{Code says:} \texttt{Z @ X - X @ Z} $\to$ $\begin{psmallmatrix}0&-2\\2&0\end{psmallmatrix} = 2iY \neq 0$.  Because $[Z,X]\neq 0$, they share no eigenbasis: a $Z$-eigenstate is an equal superposition in the $X$-eigenbasis.  \textit{(Verify: \texttt{np.linalg.eigh(X)} returns $\mathbf{e}_\pm = (1,\pm 1)/\sqrt{2}$; then \texttt{abs(e\_plus.conj().T @ e0)**2} $\to$ \texttt{0.5}.)}}}

\medskip
\noindent\textbf{The general rule:} Two observables $M$ and $N$ can be simultaneously measured (have simultaneous definite values) if and only if $[M, N] = 0$.  This is a pure linear algebra statement: commutativity $\Leftrightarrow$ shared eigenbasis.

\section{4. The Two-Level System}
% ============================================================

\begin{definition}[Two-level system]
A \textbf{two-level system} is a quantum system whose state space is $\mathbb{C}^2$.  Whenever any observable is measured, exactly one of two eigenvalues is returned.  (In physics this is called a ``qubit'' or ``spin-$\tfrac{1}{2}$.'')
\end{definition}

\noindent The standard basis vectors are
\[
    \mathbf{e}_0 = \begin{pmatrix}1\\0\end{pmatrix}, \qquad
    \mathbf{e}_1 = \begin{pmatrix}0\\1\end{pmatrix}.
\]
Any state can be written as $\mathbf{v} = \alpha_0\,\mathbf{e}_0 + \alpha_1\,\mathbf{e}_1$ with $|\alpha_0|^2 + |\alpha_1|^2 = 1$.

\section{5. Notation Reference}
% ============================================================

This section fixes the linear algebra conventions used throughout.  If you already work with complex matrices in NumPy, this is review---it is here only to establish notation.

\medskip
\noindent\textbf{Vectors.}  A state is a column vector $\mathbf{v} \in \mathbb{C}^n$ with $\mathbf{v}^\dagger\mathbf{v} = 1$.  The \textbf{conjugate transpose} (dagger) of $\mathbf{v} = \begin{psmallmatrix}\alpha_1\\\alpha_2\end{psmallmatrix}$ is the row vector $\mathbf{v}^\dagger = \begin{psmallmatrix}\alpha_1^*&\alpha_2^*\end{psmallmatrix}$.  Key rule: $(z\mathbf{v})^\dagger = z^*\mathbf{v}^\dagger$.

\medskip
\noindent\textbf{Inner product.}  $\mathbf{u}^\dagger\mathbf{v} = \sum_k \beta_k^*\alpha_k \in \mathbb{C}$.  Conjugate symmetry: $\mathbf{u}^\dagger\mathbf{v} = (\mathbf{v}^\dagger\mathbf{u})^*$.  Squared norm: $\|\mathbf{v}\|^2 = \mathbf{v}^\dagger\mathbf{v} \geq 0$.

\medskip
\noindent\textbf{Orthogonality.}  $\mathbf{u} \perp \mathbf{v}$ iff $\mathbf{u}^\dagger\mathbf{v} = 0$.  The \textbf{dimension} of the space is the maximum number of mutually orthogonal nonzero vectors.  For a two-level system, $\dim = 2$.

\medskip
\noindent\textbf{Standard basis.}  $\mathbf{e}_0 = \begin{psmallmatrix}1\\0\end{psmallmatrix}$, $\mathbf{e}_1 = \begin{psmallmatrix}0\\1\end{psmallmatrix}$.  Any $\mathbf{v} \in \mathbb{C}^2$ decomposes as $\mathbf{v} = \alpha_0\,\mathbf{e}_0 + \alpha_1\,\mathbf{e}_1$ with $|\alpha_0|^2 + |\alpha_1|^2 = 1$.

\medskip
\noindent\textit{For a two-level system, the state space is $\mathbb{C}^2$.  Next time we determine the eigenvectors of the three standard observables and their explicit column-vector forms.}
